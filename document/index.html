endpoint => //boseok.iptime.org:3002/chat

        emit 보내기
        on 수신
        아래 소켓 API 싱글쿼트('') 사이에 있는 게 이벤트 이름,
        그 다음이 payload interface임. 아래쪽에 정리해놨음

        소켓연결 => emit, on, 'connect'
        소켓끊기 => emit, on, 'disconnect'
        
        room 입장 => emit, 'room', IClient,
        Response로 현재 접속해있는 멤버들을 리턴해줌 (IClient[])
        
        메시지보내기 => emit, 'message', IMessageBody
        
        누군가 방에 들어오면 트리거 => on 'newClient',
        Response로 IClient 리턴해줌.
        

        interface IMessageBody {
          senderId: string;
          roomId: string;
          date: number;
          text: string;
        }
        export interface IClient {
          roomId: string;
          nickname: string;
          socketId: string;
        }
        interface IClientMeta {
          nickname: string;
        }
